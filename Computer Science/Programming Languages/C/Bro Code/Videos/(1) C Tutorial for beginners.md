```c
//Tells compiler to include the contents of a file
//std stands for standard
//io stands for input/output
//file contains useful functions for input and output
#include <stdio.h>

//Entry point of program is main function
//Anything in main function is read procedurally
int main(){

    //escape sequence for new line character
    //The print is formatted
    printf("I like pizza! \n");
    printf("It's really good! \n");
    
    //semicolon used to terminate statements
    //returns exit status of program
    //Will return 0 if program runs successfully with no errors
    //If error, will return 1
    return 0; 
}
```
